package cdio2;

import java.util.Random;

public class Dice
{
	private int d1, d2; // Value of dice 1 and 2.
	private int eyes = 6; // Variable over the number of dice eyes.
	
	Random ran = new Random(); // Instance of the Random class.
	
	/**
	 * Uses the Random class to roll a dice.
	 * @return The sum of the two dices
	 */
	public void roll()
	{
		//To terninger defineres.
		d1 = ran.nextInt(eyes)+1;
		d2 = ran.nextInt(eyes)+1;
	}
	
	
	public int getDiceValue1() {
		return d1;
	}
	
	public int getDiceValue2() {
		return d2;
	}
	
	
	
}
