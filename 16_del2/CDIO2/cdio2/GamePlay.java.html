package cdio2;

public class GamePlay {
	int diceValue;
	int balanceChange;
	String textID;
	boolean gameWon = false;
	int dice1, dice2;
	private boolean extraTurn = false;
	
	/**
	 * getBalanceChange returnerer ændringen i spillerens balance
	 * @return
	 */
	public int getBalanceChange() {
		return balanceChange;
	}
	
	/**
	 * getGameWon returnerer gameWon
	 * @return
	 */
	public boolean getGameWon() {
		return gameWon;
	}
	
	/**
	 * setGameWon sætter gameWon til at være en boolean
	 * @param bool
	 */
	public void setGameWon(boolean bool) {
		gameWon = bool;
	}
	
	/**
	 * getExtraTurn returner extraTurn
	 * @return
	 */
	public boolean getExtraTurn() {
		return extraTurn;
	}
	
	/**
	 * useTurn anvender værdien fra dice klassen og pålægger en StringID samt en spillerændring
	 * @param dice1: Tager en int fra dice klassen
	 * @param dice2: Tager en int fra dice klassen
	 */
	public void useTurn(int dice1, int dice2)
	{
		extraTurn = false;
		this.dice1 = dice1;
		this.dice2 = dice2;
		diceValue = dice1 + dice2;
		switch (diceValue)
		{
		case 2: 
			textID = "tower";
			// Prints event based string, with the correct message.
			balanceChange = 250; // Changes the balance of the current player.
			break;
		case 3:
			textID = "crater";
			balanceChange = -100;
			break;
		case 4:
			textID = "palaceGates";
			balanceChange = 100;
			break;
		case 5:
			textID = "coldDessert";
			balanceChange = -20;
			break;
		case 6:
			textID = "walledCity";
			balanceChange = 180;
			break;
		case 7:
			textID = "monastery";
			balanceChange = 0;
			break;
		case 8:
			textID = "blackCave";
			balanceChange = -70;
			break;
		case 9:
			textID = "mountain";
			balanceChange = 60;
			break;
		case 10:
			textID = "werewall";
			balanceChange = -80;
			extraTurn = true;
			break;
		case 11:
			textID = "pit";
			balanceChange = -50;
			break;
		case 12:
			textID = "goldmine";
			balanceChange = 650;
			break;
		}
	}
}
