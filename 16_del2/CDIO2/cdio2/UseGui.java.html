package cdio2;

import java.awt.Color;
import java.util.Scanner;

import desktop_codebehind.Car;
import desktop_codebehind.Car.Builder;
import desktop_fields.Field;
import desktop_resources.GUI;

public class UseGui {
	Player[] player;
	Scanner scan = new Scanner(System.in);
	int numberOfPlayers = 2;
	String name, color;
	Car car0 = new Car.Builder()
			.primaryColor(Color.RED)
			.secondaryColor(Color.BLUE)
			.build();
	Car car1 = new Car.Builder()
			.primaryColor(Color.BLUE)
			.secondaryColor(Color.RED)
			.build();
	
	/**
	 * Danner spille pladen der skal bruges til spillet. 
	 * Denne metode skal køres først!
	 * @param fields
	 */
	public void startGame(Field[] fields) {
		GUI.create(fields);
		GUI.setDice(1, 3);
	}
	
	/**
	 * Tilføjer det antal spiller der skal spille, med navn og brik farve.
	 */
	private void assignPlayers(String namePlayer) {
		player = new Player[numberOfPlayers]; // Creates array
		/*
		 * Assigns Name and color.
		 */
		for (int i = 0; i < numberOfPlayers; i++)
		{
			name = GUI.getUserString(namePlayer + (i+1) + " : ");
			player[i] = new Player(name);
		}
	}
	
	/**
	 * Tilføjer spillerne til pladen.
	 */
	public void addPlayerToBoard(String namePlayer) {
		assignPlayers(namePlayer);
		GUI.addPlayer(player[0].getPlayerName(), player[0].getWallet().getBalance(), car0);
		GUI.addPlayer(player[1].getPlayerName(), player[1].getWallet().getBalance(), car1);
		
	}
	
	/**
	 * setText sætter displayChanceCard til String'en txt
	 * @param txt
	 */
	public void setText(String txt) {
		GUI.showMessage(txt);
	}
	
	/**
	 * getUserOK danner et felt med parametrne String msg og Stirng buttons
	 * @param msg
	 * @param buttons
	 */
	public void getUserOK(String msg, String buttons) {
		GUI.getUserButtonPressed(msg, buttons);
	}
	
	/**
	 * setDice danner to terninger der viser hhv d1 og d2
	 * @param d1
	 * @param d2
	 */
	public void setDice(int d1, int d2) {
		GUI.setDice(d1, d2);
	}
	
	/**
	 * setNewBalance danner et felt der viser en int newBalance og en String name
	 * @param name
	 * @param newBalance
	 */
	public void setNewBalance(String name, int newBalance) {
		GUI.setBalance(name, newBalance);
	}
	
	/**
	 * setPlayer sætter en figur (Bil) til en spiller og hvor på pladen spilleren er
	 * @param field
	 * @param player
	 */
	public void setPlayer(int field, String player) {
		GUI.setCar(field, player);
	}
	
	/**
	 * removePlayer fjerner en figur (Bil) af en spiller fra det angivede felt
	 * @param field
	 * @param player
	 */
	public void removePlayer(int field, String player) {
		GUI.removeCar(field, player);
	}
}
