package cdio2;

public class Wallet{
	private int balance = 1000;
	
	/**
	 * getBlance returnerer balance
	 * @return
	 */
	public int getBalance()
	{
		return balance;
	}
	/**
	 * setBalance holder scoren og sørger for at balancen ikke går i minus
	 * @param value: Summen af tidligere beløb og value
	 */
	public void setBalance(int value)
	{
		balance += value;
		if (balance < 0)
			balance = 0;
	}
}
