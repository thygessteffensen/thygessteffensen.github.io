package cdio2;

import java.util.Scanner;

	Player pPlayer; // Bliver vidst ikke brugt?

	Scanner scan = new Scanner(System.in);
	int numberOfPlayers;
	String name, color;
	
	/**
	 * Danner spille pladen der skal bruges til spillet. 
	 * Denne metode skal køres først!
	 * @param fields
	 */
	public void startGame(Field[] fields) {
		GUI.create(fields);
		GUI.setDice(1, 3);
	}
	
	/**
	 * Tilføjer det antal spiller der skal spille, med navn og brik farve.
	 */
	private void assignPlayers(String players, String namePlayer, String colorPlayer) {
		while(true)
		{
			numberOfPlayers = GUI.getUserInteger(players); // Scan user input.
			if (numberOfPlayers > 1 && numberOfPlayers < 7) // Amount of players must be 1< <7
				break;
		}
		player = new Player[numberOfPlayers]; // Creates array
		/*
		 * Assigns Name and color.
		 */
		for (int i = 0; i < numberOfPlayers; i++)
		{
			name = GUI.getUserString(namePlayer + (i+1) + " : ");
			color = GUI.getUserString(colorPlayer);
			player[i] = new Player(name, color);
		}
	}
	
	/**
	 * Tilføjer spillerne til pladen.
	 */
	public void addPlayerToBoard(String players, String namePlayer, String colorPlayer) {
		assignPlayers(players, namePlayer, colorPlayer);
		for(int i = 0; i < numberOfPlayers; i++) {
			GUI.addPlayer(player[i].getPlayerName(), player[i].getWallet().getBalance());
		}
	}
